56  	public void actionPerformed(ActionEvent e) {
57  		if (e.getSource() == comboBackground) {
58  			bLabel1.getBusyPainter().setBaseColor(colors[comboBackground.getSelectedIndex()-1]);
59  		} else if (e.getSource() == comboForeground) {
60  			bLabel1.getBusyPainter().setHighlightColor(colors[comboForeground.getSelectedIndex()-1]);
61  		} else if (e.getSource() == btnStart) {
62  			bLabel1.setBusy(true);
63  		} else if (e.getSource() == btnStop) {
64  			bLabel1.setBusy(false);
65  		}
